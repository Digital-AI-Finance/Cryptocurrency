\activitybox{Activity 1: Explore Ethereum}{10 min}{%
Visit \url{https://etherscan.io} and find any smart contract. Answer:
\begin{enumerate}
\item What is the contract's address?
\item How many transactions has it processed?
\item What is the contract's ETH balance?
\item Can you identify what the contract does from its name or verified source code?
\end{enumerate}
\textbf{Bonus:} Find a transaction to this contract. What was the gas used? What was the gas price in Gwei?
}

% ── Activity 3 ────────────────────────────────────────────────────────────────
\activitybox{Activity 3: Read a Smart Contract}{10 min}{%
Read this Solidity contract and answer the questions:
\vspace{2pt}
\noindent\ttfamily\small
// SPDX-License-Identifier: MIT\\
pragma solidity \textasciicircum{}0.8.0;\\[2pt]
contract SimpleVault \{\\
\hspace*{1.5em}mapping(address => uint256) public balances;\\[2pt]
\hspace*{1.5em}function deposit() public payable \{\\
\hspace*{3em}balances[msg.sender] += msg.value;\\
\hspace*{1.5em}\}\\[2pt]
\hspace*{1.5em}function withdraw(uint256 amount) public \{\\
\hspace*{3em}require(balances[msg.sender] >= amount);\\
\hspace*{3em}balances[msg.sender] -= amount;\\
\hspace*{3em}payable(msg.sender).transfer(amount);\\
\hspace*{1.5em}\}\\
\}
\rmfamily\normalsize\par\vspace{2pt}
\begin{enumerate}
\item What is the state variable and its type?
\item What does \texttt{msg.sender} represent?
\item What does \texttt{msg.value} represent?
\item What happens if you try to withdraw more than your balance?
\item Can you spot a potential security vulnerability? (Hint: think about the order of operations)
\end{enumerate}
}

% ── Glossary ───────────────────────────────────────────────────────────────────
\section{Key Terms}

\renewcommand{\arraystretch}{1.25}
\noindent\begin{tabular}{@{}p{2.8cm}p{11.8cm}@{}}
\toprule
\textbf{Term} & \textbf{Definition} \\
\midrule
\textbf{EVM}           & Ethereum Virtual Machine. The sandboxed runtime environment that executes smart contract bytecode on every Ethereum node. \\
\textbf{Gas}           & The unit measuring computational effort required to execute operations on the EVM. Every instruction costs a defined amount of gas. \\
\textbf{Wei}           & The smallest denomination of ether. $1\,\text{ETH} = 10^{18}\,\text{Wei}$. Used for precise on-chain arithmetic. \\
\textbf{Gwei}          & Gigawei. $1\,\text{Gwei} = 10^{9}\,\text{Wei}$. The conventional unit for expressing gas prices. \\
\textbf{Solidity}      & The most widely used high-level programming language for writing Ethereum smart contracts, syntactically similar to JavaScript/C++. \\
\textbf{ABI}           & Application Binary Interface. A JSON specification describing a contract's functions and events, enabling external callers to encode and decode calls. \\
\textbf{Bytecode}      & The low-level EVM instructions produced by compiling Solidity source code. This is what is actually stored on-chain and executed. \\
\textbf{EOA}           & Externally Owned Account. An Ethereum account controlled by a private key (a human wallet), as opposed to a contract account. \\
\textbf{Smart Contract} & Self-executing code stored on the blockchain that automatically enforces agreement terms without a trusted intermediary. \\
\textbf{Nonce}         & A per-account transaction counter. Each new transaction must increment the nonce, preventing replay attacks and ordering transactions. \\
\bottomrule
\end{tabular}
