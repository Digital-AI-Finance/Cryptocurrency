% ── Activity 1 ────────────────────────────────────────────────────────────────
\activitybox{Activity 1: Explore ERC-20 Tokens on Etherscan}{10 min}{%
Visit \url{https://etherscan.io/tokens} and pick any ERC-20 token from the top 20. Answer:
\begin{enumerate}
\item What is the token's name, symbol, and contract address?
\item What is the total supply? How many holders does it have?
\item Click the \textbf{Contract} tab --- is the source code verified? What Solidity version was used?
\item Find a recent Transfer event in the \textbf{Events} tab. Who sent tokens to whom, and how many?
\end{enumerate}
\textbf{Bonus:} Check the token's \textbf{Read Contract} tab. Call {\ttfamily\small totalSupply()} and {\ttfamily\small decimals()}. What do they return?
}

% ── Glossary ───────────────────────────────────────────────────────────────────
\section{Key Terms}

\renewcommand{\arraystretch}{1.25}
\noindent\begin{tabular}{@{}p{2.8cm}p{11.8cm}@{}}
\toprule
\textbf{Term} & \textbf{Definition} \\
\midrule
\textbf{ERC-20}       & Ethereum Request for Comments 20. The standard interface for fungible tokens, defining 6 functions and 2 events. \\
\textbf{Fungible}     & Interchangeable; one token equals any other of the same type, like dollar bills. \\
\textbf{totalSupply}  & ERC-20 function returning the total number of tokens in existence. \\
\textbf{approve}      & ERC-20 function granting a spender permission to transfer tokens on the owner's behalf. \\
\textbf{transferFrom} & ERC-20 function enabling a pre-approved spender to move tokens between addresses. \\
\textbf{OpenZeppelin} & A library of audited, secure smart contract implementations widely used in production. \\
\textbf{Decimals}     & The number of decimal places a token uses (typically 18), enabling fractional amounts. \\
\textbf{Tokenomics}   & The economic design of a token: supply model, distribution, incentives, and value capture. \\
\textbf{Vesting}      & A schedule that gradually unlocks tokens over time, preventing large dumps after launch. \\
\textbf{Minting}      & Creating new tokens, increasing total supply. Controlled by access modifiers like {\ttfamily onlyOwner}. \\
\bottomrule
\end{tabular}
